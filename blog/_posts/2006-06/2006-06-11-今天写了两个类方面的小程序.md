#include <iostream.h>
#include <string.h>
class cPerson
{
 char name[20];
 int age;
public:
 int number;
 void setname(char *ss){strcpy(name,ss);}

#include <iostream.h>
#include <string.h>
class cPerson
{
 int age;
public:
 int number;
 char name[20];
 void setage(int tage){age=(tage<0)?0:tage; }
 void disp();
};
void cPerson::disp()
{
 cout<<"Name"<<name<<'\n';
 cout<<"age"<<age<<'\n';
}
void main()
{
 
 cPerson man;
 man.number=1;
 cin>>man.name;
 man.setage(20);
 
 cPerson woman;
 woman.number=2;
 cin>>woman.name;
 woman.setage(21);
 
 man.disp();
 woman.disp();
}
